六、部署步骤
--------------------------------------------------------------------------------
步骤1: 上传部署包
  - 将 trigen-deploy-safe.tar.gz 上传到 /tmp/ 目录
  - 上传路径: /tmp/trigen-deploy-safe.tar.gz

步骤2: 解压部署包
  cd /tmp
  tar -xzvf trigen-deploy-safe.tar.gz
  mkdir -p /tmp/trigen-deploy

步骤3: 备份将被覆盖的文件（建议执行）
  mkdir -p /tmp/backup
  for f in $(cat /tmp/trigen-deploy/deploy-whitelist.txt); do
    if [ -f "/home/admin/he-wiki-rag/$f" ]; then
      mkdir -p "/tmp/backup/$(dirname $f)"
      cp "/home/admin/he-wiki-rag/$f" "/tmp/backup/$f"
    fi
  done
  echo "备份完成"

步骤4: 按白名单部署文件（只复制白名单中的文件）
  cd /home/admin/he-wiki-rag/
  
  # 逐个复制文件（不是整个目录复制）
  while IFS= read -r filepath; do
    if [[ "$filepath" == \#* ]] || [[ -z "$filepath" ]]; then
      continue  # 跳过注释和空行
    fi
    if [ -f "/tmp/trigen-deploy/$filepath" ]; then
      mkdir -p "$(dirname $filepath)"
      cp "/tmp/trigen-deploy/$filepath" "$filepath"
      echo "已更新: $filepath"
    fi
  done < /tmp/trigen-deploy/deploy-whitelist.txt

步骤5: 安装新依赖
  cd /home/admin/he-wiki-rag/
  npm install --legacy-peer-deps --no-audit --no-fund

步骤6: 重启服务
  pm2 restart trigen
  sleep 2
  pm2 status

步骤7: 验证部署
  # 检查服务状态
  pm2 status
  
  # 检查日志
  pm2 logs trigen --lines 20 --nostream
  
  # 访问测试
  curl -s http://localhost:3000 | head -c 200

步骤8: 清理临时文件
  rm -rf /tmp/trigen-deploy*
  rm -rf /tmp/backup
